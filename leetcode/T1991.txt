int findMiddleIndex(int* nums, int numsSize) {
    int SumAll = 0;
    int res = 0;
    for(int i = 0; i < numsSize;i++){
        SumAll = SumAll + *(nums + i);
    }
    int SumNow = 0;
    for(int j =0; j<numsSize; j++){
        if(2 * SumNow + *(nums + j) == SumAll){
            return j;
        }
        else{
            SumNow = SumNow + *(nums + j);
        }            
    }
    return -1;
}
